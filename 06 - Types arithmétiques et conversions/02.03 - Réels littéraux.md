# Littéraux constants

Pour chacune des lignes de code suivante, indiquez 
- si elle compile
- si oui
  - quel est le type de la variable
  - ce que la ligne affiche; 
  
  ~~~
  auto v1 = 1.5;        cout << v1 << endl;
  
  auto v2 = 1E3;        cout << v2 << endl;
  
  auto v3 = 12u;        cout << v3 << endl;
  
  auto v4 = 12.0u;      cout << v4 << endl;
  
  auto v5 = 1L;         cout << v5 << endl;
  
  auto v6 = 1.0L;       cout << v6 << endl;
  
  auto v7 = .5;         cout << v7 << endl;
  
  auto v8 = 5.;         cout << v8 << endl;
  
  auto v9 = 100'000'000'000; cout << v9 << endl;
  
  auto v10 = 0x33;      cout << v10 << endl;
  
  auto v11 = 0xefg;     cout << v11 << endl;
  
  auto v12 = 0xef;      cout << v12 << endl;
  
  auto v13 = 0xEF;      cout << v13 << endl;
  
  auto v14 = 0x0.2;     cout << v14 << endl;
  
  auto v15 = 08;        cout << v15 << endl;
  
  auto v16 = 027;       cout << v16 << endl;
  
  auto v17 = 0x1.p0;    cout << v17 << endl;
    
  auto v18 = 0x1.8p+0f; cout << v18 << endl;
    
  auto v19 = 0x1.p-2L;  cout << v19 << endl;
    
  auto v20 = 0x1.1p+2;  cout << v20 << endl; 
  ~~~

<summary>Solution</summary>

~~~cpp
double v1 = 1.5;            // 1.5

double v2 = 1E3;            // 1000

unsigned int v3 = 12u;      // 12

// auto v4 = 12.0u;         un réel ne peut être non signé

long int v5 = 1L;           // 1

long double v6 = 1.0L;      // 1

double v7 = .5;             // .5

double v8 = 5.;             // 5

long v9 = 100'000'000'000;  // ou int ou long long, le plus petit
                            // capable de stoquer la valeur
                            // 100000000000
                            
int v10 = 0x33;             // 51

// auto v11 = 0xefg;        // g n'est pas un chiffre hexadécimal

int v12 = 0xef;             // 239

int v13 = 0xEF;             // 239

//  auto v14 = 0x0.2;         pas le bon format hexfloat

//  auto v15 = 08;            8 n'est pas un chiffre octal

int v16 = 027;              // 23

double v17 = 0x1.p0;        // 1

float v18 = 0x1.8p+0f;      // 1.5

long double v19 = 0x1.p-2L; // 0.25

double v20 = 0x1.1p+2;      // 4.25
~~~
